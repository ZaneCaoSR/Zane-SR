### 1.2 photos 表（照片元数据）

```javascript
// 表名: photos
// 描述: 存储宝宝照片的元数据信息

{
  _id: ObjectId,              // 小程序云数据库自动生成
  _openid: String,           // 创建者 OpenID（云开发自动添加）
  
  // === 基础信息 ===
  fileID: String,            // 云存储文件 ID
  cloudPath: String,         // 云存储路径，如 photos/2026/03/xxx.jpg
  width: Number,             // 图片宽度
  height: Number,            // 图片高度
  size: Number,              // 文件大小（字节）
  
  // === 时间信息 ===
  shotAt: Date,              // 拍照时间（用户可手动调整）
  createdAt: Date,           // 上传时间（系统自动）
  updatedAt: Date,           // 更新时间（系统自动）
  
  // === AI 识别结果 ===
  aiResult: {
    // 表情识别
    expression: {
      type: String,         // main: 主导表情，如 "smile", "laugh", "surprise", "calm"
      score: Number,         // 置信度 0-1
      detail: String         // 详细描述
    },
    
    // 动作识别
    action: {
      type: String,         // main: 主导动作，如 "head_up", "roll_over", "crawl", "sit", "stand"
      score: Number,        // 置信度 0-1
      detail: String        // 详细描述
    },
    
    // AI 生成描述
    description: String,     // AI 生成的图片描述（自然语言）
    tags: [String],          // 标签数组，如 ["微笑", "抬头", "可爱", "室内"]
    
    // 识别元数据
    model: String,           // 使用的 AI 模型，如 "aliyun-infant-expression-v1"
    recognizedAt: Date,      // 识别完成时间
  },
  
  // === 用户编辑 ===
  remark: String,            // 用户备注
  isMilestone: Boolean,      // 是否为里程碑照片
  milestoneType: String,     // 里程碑类型：first_smile, first_crawl, first_step 等
  
  // === 状态管理 ===
  status: String,            // processing | ready | failed
  errorMsg: String,         // 错误信息（如果有）
}
```

## 2. AI 服务集成方案

### 2.1 阿里云视觉智能接入

#### 2.1.1 服务开通

1. 访问 [阿里云视觉智能开放平台](https://vision.aliyun.com/)
2. 注册/登录阿里云账号
3. 开通「通用图像分析」或「人脸识别」服务
4. 创建 AccessKey（保存 AccessKeyId 和 AccessKeySecret）

#### 2.1.2 推荐 API

| API | 用途 | 说明 |
|-----|------|------|
| ** RecognizeInfantFace** | 婴儿表情识别 | 推荐！专门针对婴儿表情优化 |
| ** DescribeImage** | 图像描述 | 生成自然语言图片描述 |
| ** RecognizeFace** | 人脸检测 | 获取人脸位置、年龄等信息 |

> MVP 阶段推荐组合：`RecognizeInfantFace` + `DescribeImage`

#### 2.1.3 云函数设计

```javascript
// 云函数: ai-recognize
// 用途: 调用阿里云视觉智能 API 识别照片

const cloud = require('wx-server-sdk')
const Core = require('@alicloud/pop-core')  // 阿里云 SDK

cloud.init()

// 阿里云配置（建议使用云函数环境变量）
const ALIYUN_CONFIG = {
  accessKeyId: process.env.ALIYUN_ACCESS_KEY_ID,
  accessKeySecret: process.env.ALIYUN_ACCESS_KEY_SECRET,
  region: 'cn-shanghai',
  endpoint: 'https://vision.cn-shanghai.aliyuncs.com',
  apiVersion: '2020-03-30'
}

exports.main = async (event, context) => {
  const { fileID, cloudPath } = event
  
  try {
    // 1. 获取云存储中的图片（转为 Base64）
    const fileBuffer = await cloud.downloadFile({
      fileID: fileID
    })
    const base64Image = fileBuffer.fileContent.toString('base64')
    
    // 2. 调用阿里云婴儿表情识别 API
    const client = new Core(ALIYUN_CONFIG)
    
    const infantFaceResult = await client.request('RecognizeInfantFace', {
      ImageURL: `data:image/jpeg;base64,${base64Image}`
    }, {
      method: 'POST'
    })
    
    // 3. 调用图像描述 API
    const describeResult = await client.request('DescribeImage', {
      ImageURL: `data:image/jpeg;base64,${base64Image}`
    }, {
      method: 'POST'
    })
    
    // 4. 解析并组合结果
    const result = {
      expression: parseExpression(infantFaceResult),
      action: inferAction(infantFaceResult, describeResult),
      description: describeResult.Description || '宝宝的一天',
      tags: generateTags(infantFaceResult, describeResult),
      model: 'aliyun-vision-v2',
      recognizedAt: new Date()
    }
    
    // 5. 更新数据库
    await cloud.database().collection('photos')
      .where({ fileID })
      .update({
        data: {
          'aiResult': result,
          'status': 'ready',
          'updatedAt': new Date()
        }
      })
    
    return { success: true, result }
    
  } catch (error) {
    console.error('AI 识别失败:', error)
    
    // 更新状态为失败
    await cloud.database().collection('photos')
      .where({ fileID })
      .update({
        data: {
          'status': 'failed',
          'errorMsg': error.message,
          'updatedAt': new Date()
        }
      })
    
    return { success: false, error: error.message }
  }
}
```

### 2.2 API 调用流程

```
┌─────────────────────────────────────────────────────────────────┐
│                        照片上传流程                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  ┌──────────┐                                                  │
│  │ 小程序   │                                                  │
│  │ 上传照片 │                                                  │
│  └────┬─────┘                                                  │
│       │                                                        │
│       ▼                                                        │
│  ┌──────────┐     ┌──────────┐     ┌──────────┐              │
│  │ 云存储   │────▶│ 创建记录 │────▶│ 触发 AI  │              │
│  │ 保存照片 │     │ photos   │     │ 识别云函数│              │
│  └──────────┘     └──────────┘     └────┬─────┘              │
│                                          │                     │
│                                          ▼                     │
│  ┌──────────┐     ┌──────────┐     ┌──────────┐              │
│  │ 更新记录 │◀────│ 解析结果  │◀────│ 阿里云   │              │
│  │ aiResult │     │ 保存标签  │     │ 视觉智能 │              │
│  └──────────┘     └──────────┘     └──────────┘              │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

**详细步骤**：

1. **小程序端**：调用 `wx.chooseImage` 或 `wx.camera` 获取照片
2. **上传云存储**：调用 `wx.cloud.uploadFile` 上传到云存储
3. **创建数据库记录**：在 `photos` 表创建记录，`status: 'processing'`
4. **触发云函数**：调用 `ai-recognize` 云函数开始识别
5. **AI 识别**：
   - 调用 `RecognizeInfantFace` 获取表情
   - 调用 `DescribeImage` 获取描述
6. **更新数据库**：将识别结果写入 `aiResult` 字段
7. **小程序轮询/订阅**：前端通过 `wx.cloud.database().watch` 监听变化

### 2.3 错误处理方案

#### 2.3.1 错误类型分类

| 错误类型 | 原因 | 处理方案 |
|----------|------|----------|
| 网络错误 | 网络不稳定 | 自动重试 3 次，间隔 2s |
| API 配额 | 阿里云额度用完 | 提示用户稍后再试 |
| 图片格式 | 不支持的格式 | 前端拦截，只支持 jpg/png |
| 图片过大 | 超过 5MB | 前端压缩后再上传 |
| 识别失败 | AI 无法识别 | 标记 `status: failed`，允许用户手动填写 |

#### 2.3.2 重试机制（云函数）

```javascript
async function callWithRetry(fn, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn()
    } catch (error) {
      if (i === maxRetries - 1) throw error
      await new Promise(resolve => setTimeout(resolve, 2000 * (i + 1)))
    }
  }
}
```

#### 2.3.3 前端错误提示

```javascript
// 小程序端 - 处理识别结果
wx.cloud.database().collection('photos')
  .doc(photoId)
  .watch({
    onChange: function(snapshot) {
      if (snapshot.doc.status === 'ready') {
        // 识别成功，展示结果
        this.setData({ aiResult: snapshot.doc.aiResult })
      } else if (snapshot.doc.status === 'failed') {
        // 识别失败，提供手动填写
        wx.showModal({
          title: '识别失败',
          content: '无法自动识别，您可以手动填写描述',
          confirmText: '去填写',
          success: (res) => {
            if (res.confirm) {
              // 跳转到手动编辑页面
            }
          }
        })
      }
    },
    onError: function(error) {
      console.error('监听失败', error)
    }
  })
```
